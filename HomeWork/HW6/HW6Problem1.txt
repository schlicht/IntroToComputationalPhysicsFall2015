      Program Problem1
      Implicit None
      Integer :: n, bins

      Call transform(n, bins)
      Call rejection(n, bins)

      Contains
      Subroutine transform (n, bins)
        Implicit None
        Integer, Intent(in) :: n, bins
        Integer, Parameter :: seed=86456
        Real*8, Dimension(1:bins) :: xbins
        Integer :: i
        Real*8 :: x, y, p
        
        open(unit=10, file="binstrans.dat")

        Call srand(seed)
        Do i=1, bins
          xbins(i)=0
        End do

        Do i=1, n
          y=rand(seed)
          x=(DBLE(8)*y)**(1/DBLE(3))   
          p=INT(DBLE(bins/2)*x)
          xbins(p)=xbins(p)+1
        End do
        
        Do i=1, bins
          Write(10,*) i, xbins(i)
        End do
         
      End subroutine transform

      Subroutine rejection (n, bins)
        Implicit None
        Integer, Intent(in) :: n, bins
        Integer, Parameter :: seed=86456
        Real*8, Dimension(1:bins) :: xbins
        Integer :: i
        Real*8 :: x, y, p
        
        open(unit=11, file="binsrej.dat")

        Call srand(seed)
        Do i=1, bins
          xbins(i)=0
        End do

        Do while (i<n)
          y=2*rand(seed)
          x= 2*rand(seed)
          if(y<3/8*(x**2)) then
            i=i+1
            p=INT(DBLE(bins/2)*x)
            xbins(p)=xbins(p)+1
        End do
        
        Do i=1, bins
          Write(11,*) i, xbins(i)
        End do
         
      End subroutine rejection

      End Program Problem1
